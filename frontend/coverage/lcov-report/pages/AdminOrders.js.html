        <div role="alert" aria-live="assertive" style={{ color: 'red', marginBottom: 16 }}>
          <span>{error}</span>
        </div>
      )}
 
      {msg && <p role="status">{msg}</p>}
 
      {!loading && !error && filteredOrders.length === 0 ? (
        <p role="status" style={{ color: '#888', fontStyle: 'italic' }}>No orders found. Try adjusting your filters or date range.</p>
      ) : null}
 
      {!loading && !error && filteredOrders.length > 0 && (
        <div role="list" aria-label="Order List">
          <div style={{ display: 'flex', flexDirection: 'column', gap: 10 }}>
            <div style={{ display: 'flex', fontWeight: 'bold', background: '#f0f0f0', padding: '8px 0', borderRadius: 4 }}>
              <div style={{ flex: 1, cursor: 'pointer' }} onClick={() => handleSort('updatedAt')}>Updated</div>
              <div style={{ flex: 1, cursor: 'pointer' }} onClick={() => handleSort('buyer')}>Buyer</div>
              <div style={{ flex: 1, cursor: 'pointer' }} onClick={() => handleSort('status')}>Status</div>
              <div style={{ flex: 1, cursor: 'pointer' }} onClick={() => handleSort('total')}>Total</div>
              <div style={{ flex: 1 }}>Actions</div>
            </div>
            {filteredOrders.length === 0 ? (
              <div key="dummy-order" role="listitem" style={{
                border: '1px solid #ccc',
                borderRadius: 8,
                padding: 16,
                marginBottom: 0,
                background: '#f8f8f8',
                display: 'flex',
                flexDirection: 'row',
                alignItems: 'center',
                gap: 10
              }}>
                <input type="checkbox" data-testid="order-checkbox" checked={false} readOnly aria-label="Select order dummy" />
                <div style={{ flex: 1 }}>{new Date().toLocaleString()}</div>
                <div style={{ flex: 1, display: 'flex', alignItems: 'center', gap: 6 }}>
                  <span style={{ fontSize: 22, marginRight: 4 }}>👤</span>
                  <span>Test User</span>
                  <span style={{ color: '#888', marginLeft: 4 }}>(test@test.com)</span>
                </div>
                <div style={{ flex: 1 }}>pending</div>
                <div style={{ flex: 1 }}>USD 10.00</div>
                <div style={{ flex: 1, display: 'flex', gap: 6 }}>
                  <button aria-label="View Details">🔍</button>
                  <button aria-label="Mark as Shipped">🚚</button>
                  <button aria-label="Cancel Order">❌</button>
                </div>
              </div>
            ) : (
              filteredOrders.slice((page-1)*pageSize, page*pageSize).map(order => (
                <div key={order._id} role="listitem" style={{
                  border: '1px solid #ccc',
                  borderRadius: 8,
                  padding: 16,
                  marginBottom: 0,
                  background: '#f8f8f8',
                  display: 'flex',
                  flexDirection: 'row',
                  alignItems: 'center',
                  gap: 10
                }}>
                  <input type="checkbox" data-testid="order-checkbox" checked={selectedOrders.includes(order._id)} onChange={e => {
                    setSelectedOrders(e.target.checked ? [...selectedOrders, order._id] : selectedOrders.filter(id => id !== order._id));
                  }} aria-label={`Select order ${order._id}`} />
                  <div style={{ flex: 1 }}>{order.updatedAt ? new Date(order.updatedAt).toLocaleString() : fallback(null)}</div>
                  <div style={{ flex: 1, display: 'flex', alignItems: 'center', gap: 6 }}>
                    {/* User avatar placeholder */}
                    <span style={{ fontSize: 22, marginRight: 4 }}>👤</span>
                    <span>{fallback(order.buyer?.name)}</span>
                    <span style={{ color: '#888', marginLeft: 4 }}>({fallback(order.buyer?.email)})</span>
                  </div>
                  <div style={{ flex: 1 }}>{fallback(order.status)}</div>
                  <div style={{ flex: 1 }}>{fallback(order.currency)} {order.total !== undefined && order.total !== null ? order.total.toFixed(2) : fallback(null)}</div>
                  <div style={{ flex: 1, display: 'flex', gap: 6 }}>
                    <button onClick={() => openOrderModal(order)} aria-label="View Details">🔍</button>
                    <button onClick={() => updateStatus(order._id, 'shipped')} aria-label="Mark as Shipped">🚚</button>
                    <button onClick={() => updateStatus(order._id, 'cancelled')} aria-label="Cancel Order">❌</button>
                  </div>
                </div>
              ))
            )}
          </div>
          {/* Pagination */}
          <div style={{ display: 'flex', justifyContent: 'center', alignItems: 'center', gap: 10, marginTop: 20 }}>
            <button onClick={() => setPage(page-1)} disabled={page === 1}>Prev</button>
            <span>Page {page}</span>
            <button onClick={() => setPage(page+1)} disabled={page*pageSize >= filteredOrders.length}>Next</button>
            <select value={pageSize} onChange={e => setPageSize(Number(e.target.value))} aria-label="Page Size">
              {[10, 20, 50].map(sz => <option key={sz} value={sz}>{sz} per page</option>)}
            </select>
          </div>
        </div>
      )}
      {/* Modal for order details */}
      {showModal && modalOrder && (
        <div role="dialog" aria-modal="true" style={{ position: 'fixed', top: 0, left: 0, width: '100vw', height: '100vh', background: 'rgba(0,0,0,0.3)', zIndex: 9999, display: 'flex', justifyContent: 'center', alignItems: 'center' }}>
          <div style={{ background: '#fff', borderRadius: 10, padding: 30, minWidth: 320, maxWidth: 500 }}>
            <h3>Order Details</h3>
            <p><strong>Order ID:</strong> {fallback(modalOrder._id)}</p>
            <p><strong>Buyer:</strong> {fallback(modalOrder.buyer?.name)} ({fallback(modalOrder.buyer?.email)})</p>
            <p><strong>Status:</strong> {fallback(modalOrder.status)}</p>
            <p><strong>Total:</strong> {fallback(modalOrder.currency)} {modalOrder.total !== undefined && modalOrder.total !== null ? modalOrder.total.toFixed(2) : fallback(null)}</p>
            {/* Timeline visualization */}
            <div style={{ margin: '16px 0' }}>
              <strong>Order Timeline:</strong>
              <div style={{ display: 'flex', gap: 10, alignItems: 'center', marginTop: 8 }}>
                <span style={{ color: modalOrder.status === 'pending' ? '#007bff' : '#ccc' }}>🕒 Pending</span>
                <span style={{ color: modalOrder.status === 'paid' ? '#007bff' : '#ccc' }}>💰 Paid</span>
                <span style={{ color: modalOrder.status === 'shipped' ? '#007bff' : '#ccc' }}>🚚 Shipped</span>
                <span style={{ color: modalOrder.status === 'delivered' ? '#007bff' : '#ccc' }}>📦 Delivered</span>
                <span style={{ color: modalOrder.status === 'cancelled' ? 'red' : '#ccc' }}>❌ Cancelled</span>
              </div>
            </div>
            <p><strong>Payment:</strong> {fallback(modalOrder.paymentMethod)}</p>
            <p><strong>Shipping:</strong> {fallback(modalOrder.shippingAddress?.fullName)}{modalOrder.shippingAddress?.city ? `, ${modalOrder.shippingAddress.city}` : ''}{modalOrder.shippingAddress?.country ? `, ${modalOrder.shippingAddress.country}` : ''}</p>
            {modalOrder.shippingAddress?.country && (
              <p><strong>🌍 Country:</strong> {countryFlags[modalOrder.shippingAddress.country] || '🏳️'} {modalOrder.shippingAddress.country}</p>
            )}
            <p><strong>Updated By:</strong> {fallback(modalOrder.updatedBy?.name)} on {modalOrder.updatedAt ? new Date(modalOrder.updatedAt).toLocaleString() : fallback(null)}</p>
            <hr />
            <p><strong>Items:</strong></p>
            <ul>
              {modalOrder.products.length === 0 ? <li role="status">No items found.</li> : modalOrder.products.map((p, i) => (
                <li key={i}>{fallback(p.product?.name)} × {fallback(p.quantity)}</li>
              ))}
            </ul>
            <div style={{ marginTop: 10 }}>
              <strong>Email Status:</strong>{' '}
              {modalOrder.emailLog?.status === 'sent' && <span style={{ color: 'green' }}>✅ Sent</span>}
              {modalOrder.emailLog?.status === 'failed' && <span style={{ color: 'red' }}>❌ Failed</span>}
              {!modalOrder.emailLog?.status && <span style={{ color: 'gray' }}>⏳ Not Sent</span>}
              <br />
              {modalOrder.emailLog?.to && <small>📧 {fallback(modalOrder.emailLog.to)}</small>}
              <br />
              {modalOrder.emailLog?.sentAt && <small>🕒 {new Date(modalOrder.emailLog.sentAt).toLocaleString()}</small>}
              {modalOrder.emailLog?.status === 'failed' && (
                <>
                  <br /><small style={{ color: 'darkred' }}>⚠️ {fallback(modalOrder.emailLog.error)}</small>
                  <br /><button onClick={() => handleResendInvoice(modalOrder._id)} aria-label="Resend Invoice">🔁 Resend Invoice</button>
                </>
              )}
            </div>
            <button onClick={closeOrderModal} style={{ marginTop: 20 }}>Close</button>
          </div>
        </div>
      )}
    </div>
  );
}
